import cv2  # å›¾ç‰‡å¤„ç†ä¸‰æ–¹åº“ï¼Œç”¨äºå¯¹å›¾ç‰‡è¿›è¡Œå‰åå¤„ç†
import numpy as np  # ç”¨äºå¯¹å¤šç»´æ•°ç»„è¿›è¡Œè®¡ç®—
import torch  # æ·±åº¦å­¦ä¹ è¿ç®—æ¡†æ¶ï¼Œæ­¤å¤„ä¸»è¦ç”¨æ¥å¤„ç†æ•°æ®

from mindx.sdk import Tensor  # mxVision ä¸­çš„ Tensor æ•°æ®ç»“æ„

# æ¨¡å‹å‰åå¤„ç†ç›¸å…³å‡½æ•°
from det_utils import get_labels_from_txt, letterbox, scale_coords, nms, draw_bbox

# jupyter æ˜¾ç¤ºç”¨
import ipywidgets as widgets

# å˜é‡åˆå§‹åŒ–

DEVICE_ID = 0  # è®¾å¤‡id

# image_path = 'world_cup.jpg'  # æµ‹è¯•å›¾ç‰‡è·¯å¾„

# æŸ¥æ‰¾ USB camera çš„ index å€¼
def find_camera_index():
    max_index_to_check = 10  # Maximum index to check for camera

    for index in range(max_index_to_check):
        cap = cv2.VideoCapture(index)
        if cap.read()[0]:
            cap.release()
            return index

    # If no camera is found
    raise ValueError("No camera found.")


#
def img2bytes(image):
    """å°†å›¾ç‰‡è½¬æ¢ä¸ºå­—èŠ‚ç """
    return bytes(cv2.imencode(".jpg", image)[1])


# è·å–æ‘„åƒå¤´
camera_index = find_camera_index()
cap = cv2.VideoCapture(camera_index)

# è·å–ä¿å­˜è§†é¢‘ç›¸å…³å˜é‡
fps = 5  # ä½¿ç”¨rtspæ¨æµæ—¶ï¼Œä¸èƒ½ä½¿ç”¨cap.get(cv2.CAP_PROP_FPS)æ¥è·å–å¸§ç‡ï¼Œä¸”ç”±äºå»¶è¿Ÿè¾ƒé«˜ï¼Œæ‰‹åŠ¨æŒ‡å®šå¸§ç‡ï¼Œå¯ä»¥æ ¹æ®å®é™…æƒ…å†µè°ƒèŠ‚
fourcc = cv2.VideoWriter_fourcc(*"mp4v")

outfile = "video_result.mp4"
video_width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
video_height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
writer = cv2.VideoWriter(outfile, fourcc, fps, (video_width, video_height))

# jupyter åˆå§‹åŒ–è§†é¢‘æ˜¾ç¤ºç•Œé¢
image_widget = widgets.Image(format="jpeg", width=video_width, height=video_height)
display(image_widget)

try:
    while cap.isOpened():  # åœ¨æ‘„åƒå¤´æ‰“å¼€çš„æƒ…å†µä¸‹å¾ªç¯æ‰§è¡Œ
        ret, frame = cap.read()  # æ­¤å¤„ frame ä¸º bgr æ ¼å¼å›¾ç‰‡

        # æ•°æ®å‰å¤„ç†
        # img_bgr = cv2.imread(image_path, cv2.IMREAD_COLOR)  # è¯»å…¥å›¾ç‰‡
        # img, scale_ratio, pad_size = letterbox(img_bgr, new_shape=[640, 640])  # å¯¹å›¾åƒè¿›è¡Œç¼©æ”¾ä¸å¡«å……ï¼Œä¿æŒé•¿å®½æ¯”
        img, scale_ratio, pad_size = letterbox(
            frame, new_shape=[640, 640]
        )  # å¯¹å›¾åƒè¿›è¡Œç¼©æ”¾ä¸å¡«å……ï¼Œä¿æŒé•¿å®½æ¯”
        img = img[:, :, ::-1].transpose(2, 0, 1)  # BGR to RGB, HWC to CHW
        img = np.expand_dims(img, 0).astype(
            np.float32
        )  # å°†å½¢çŠ¶è½¬æ¢ä¸º channel first (1, 3, 640, 640)ï¼Œå³æ‰©å±•ç¬¬ä¸€ç»´ä¸º batchsize
        img = np.ascontiguousarray(img) / 255.0  # è½¬æ¢ä¸ºå†…å­˜è¿ç»­å­˜å‚¨çš„æ•°ç»„
        img = Tensor(img)  # å°†numpyè½¬ä¸ºè½¬ä¸ºTensorç±»

        # æ¨¡å‹æ¨ç†, å¾—åˆ°æ¨¡å‹è¾“å‡º
        # åˆå§‹åŒ– base.model ç±»
        model = base.model(modelPath=model_path, deviceId=DEVICE_ID)
        # æ‰§è¡Œæ¨ç†ã€‚è¾“å…¥æ•°æ®ç±»å‹ï¼šList[base.Tensor]ï¼Œ è¿”å›æ¨¡å‹æ¨ç†è¾“å‡ºçš„ List[base.Tensor]
        output = model.infer([img])[0]

        # åå¤„ç†
        output.to_host()  # å°† Tensor æ•°æ®è½¬ç§»åˆ° Host ä¾§
        output = np.array(output)  # å°†æ•°æ®è½¬ä¸º numpy array ç±»å‹
        # åˆ©ç”¨éæå¤§å€¼æŠ‘åˆ¶å¤„ç†æ¨¡å‹è¾“å‡ºï¼Œconf_thres ä¸ºç½®ä¿¡åº¦é˜ˆå€¼ï¼Œiou_thres ä¸ºioué˜ˆå€¼
        boxout = nms(torch.tensor(output), conf_thres=0.4, iou_thres=0.5)
        pred_all = boxout[0].numpy()  # è½¬æ¢ä¸ºnumpyæ•°ç»„
        # å°†æ¨ç†ç»“æœç¼©æ”¾åˆ°åŸå§‹å›¾ç‰‡å¤§å°
        scale_coords(
            [640, 640], pred_all[:, :4], frame.shape, ratio_pad=(scale_ratio, pad_size)
        )
        # å¾—åˆ°ç±»åˆ«ä¿¡æ¯ï¼Œè¿”å›åºå·ä¸ç±»åˆ«å¯¹åº”çš„å­—å…¸
        labels_dict = get_labels_from_txt("./coco_names.txt")
        # ç”»å‡ºæ£€æµ‹æ¡†ã€ç±»åˆ«ã€æ¦‚ç‡
        img_dw = draw_bbox(pred_all, frame, (0, 255, 0), 2, labels_dict)

        # å°†æ¨ç†ç»“æœå†™å…¥è§†é¢‘
        writer.write(img_dw)

        # å°†æ¨ç†ç»“æœæ˜¾ç¤ºåœ¨jupyter
        image_widget.value = img2bytes(img_dw)


except KeyboardInterrupt:
    cap.release()
    writer.release()
finally:
    cap.release()
    writer.release()
# ä¿å­˜å›¾ç‰‡åˆ°æ–‡ä»¶
print("save infer result success")
